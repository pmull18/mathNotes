\chapter{Countability}


\dfn[]{}{
	A set $X$ is at most \textbf{countable} if we can enumerate the points of $X$ with finitely many elements or a sequence i.e. $X$ is finite: $\exists k\in\mathbb{Z}_{\geq 0}$ and $x_1,...,x_k$ such that $X=\{x_1,...,x_k\}$ or $\exists (x_n)_{n\geq 1}\subset X$ such that $X = \{x_n: n\geq 1\}$.
}

\ex[]{}{
	$\mathbb{Z}_1$ is countable, $\mathbb{Q} = \{\frac{p}{q}:p\in\mathbb{Z}, q\in\mathbb{Z}_{\geq 1}\}$ is countable, 
}
